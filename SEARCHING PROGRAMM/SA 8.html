     int binarySearch(char arr[], int n, char key) {
          int l = 0, u = n-1;
          while(l <= u)
          {
              int mid = l + (u - l) / 2;
              if (arr[mid] == key)
              {
                  return mid;
              }
              else if(arr[mid] > key){
                  u = mid-1;
              }
              else {
                  l = mid+1;
              }
          }
          return -1; 
      }
